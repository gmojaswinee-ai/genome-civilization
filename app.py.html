import random

import gradio as gr


bases = ["A", "T", "G", "C"]


# -------------------------

# CORE FUNCTIONS

# -------------------------


def mutate(seq, p):

    seq = list(seq)

    for i in range(len(seq)):

        if random.random() < p:

            seq[i] = random.choice(bases)

    return "".join(seq)


def fitness(dna):

    return (dna.count("G") + dna.count("C")) / len(dna)


def classify(gc):

    if gc > 0.6:

        return "ORDER"

    elif gc > 0.45:

        return "BALANCE"

    else:

        return "CHAOS"


def generate_myth(world_name, state, step):

    if state == "ORDER":

        return f"[{world_name} Step {step}] Stability becomes sacred."

    elif state == "CHAOS":

        return f"[{world_name} Step {step}] Mutation breaks the ancient code."

    else:

        return f"[{world_name} Step {step}] The system balances between forces."


# -------------------------

# WORLDS

# -------------------------


worlds = [

    {

        "name": "WORLD A",

        "pressure": 0.2,

        "pop": ["ATGCGTACGTTAGC", "GCTATGCGTACGTA"],

        "history": {"ORDER": [], "BALANCE": [], "CHAOS": []},

        "myth": []

    },

    {

        "name": "WORLD B",

        "pressure": 0.5,

        "pop": ["TTAGCGCATGGCAT", "CGTACGTAGCTTAC"],

        "history": {"ORDER": [], "BALANCE": [], "CHAOS": []},

        "myth": []

    },

    {

        "name": "WORLD C",

        "pressure": 0.8,

        "pop": ["ATATGCGCGTATGC", "GCGTACGATCGTAC"],

        "history": {"ORDER": [], "BALANCE": [], "CHAOS": []},

        "myth": []

    }

]


# -------------------------

# SIMULATION

# -------------------------


def run_simulation(steps):


    output = ""


    for step in range(int(steps)):


        output += f"\n===== STEP {step} =====\n"


        for w in worlds:


            new_pop = []

            order = balance = chaos = 0


            for dna in w["pop"]:


                dna = mutate(dna, w["pressure"])

                gc = fitness(dna)

                state = classify(gc)


                if state == "ORDER":

                    order += 1

                elif state == "BALANCE":

                    balance += 1

                else:

                    chaos += 1


                new_pop.append(dna)


            w["pop"] = new_pop


            w["history"]["ORDER"].append(order)

            w["history"]["BALANCE"].append(balance)

            w["history"]["CHAOS"].append(chaos)


            dominant = max(

                [("ORDER", order), ("BALANCE", balance), ("CHAOS", chaos)],

                key=lambda x: x[1]

            )[0]


            myth = generate_myth(w["name"], dominant, step)

            w["myth"].append(myth)


            output += f"""

{w['name']}

ORDER:{order} BALANCE:{balance} CHAOS:{chaos}

MYTH: {myth}

-------------------

"""


    return output


# -------------------------

# UI

# -------------------------


demo = gr.Interface(

    fn=run_simulation,

    inputs=gr.Slider(1, 20, value=5, step=1),

    outputs=gr.Textbox(),

    title="Genome Civilization Simulator"

)


demo.launch()
